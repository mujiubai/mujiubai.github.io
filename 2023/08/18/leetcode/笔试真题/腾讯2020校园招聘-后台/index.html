1：压缩算法

小Q想要给他的朋友发送一个神秘字符串，但是他发现字符串的过于长了，于是小Q发明了一种压缩算法对字符串中重复的部分进行了压缩，对于字符串中连续的m个相同字符串S将会压缩为m|S，例如字符串ABCABCABC将会被压缩为[3|ABC]，现在小Q的同学收到了小Q发送过来的字符串，你能帮助他进行解压缩么？

不断压入栈，需要注意细节即可。

  • 时间复杂度：O()
  • 空间复杂度：O()
class Solution {
public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 
     * @param str string字符串 
     * @return string字符串
     */
    string compress(string str) {
        // write code here
        string ans;
        int i=0;
        stack<int> st1;
        stack<char> st2;
        while(i<str.size()){
            if(str[i]=='['){
                ++i;
                int j=i+1;
                while(str[j]!='|') ++j;
                st1.push(atoi(str.substr(i,j-i).c_str()));
                i=j+1;                
                st2.push('[');
            }else if(str[i]==']'){
                string tmp;
                while(!st2.empty()&&st2.top()!='['){
                    tmp+=st2.top();
                    st2.pop();
                }
                st2.pop();
                reverse(tmp.begin(),tmp.end());
                int num=st1.top();
                st1.pop();
                for(int k=0;k<num;++k){
                    for(int j=0;j<tmp.size();++j){
                        st2.push(tmp[j]);
                    }                    
                }
                ++i;
            }else{
                st2.push(str[i]);
                ++i;
            }          
        }
        while(!st2.empty()){
            ans+=st2.top();
            st2.pop();
        }
        reverse(ans.begin(),ans.end());
        return ans;
    }
};

2：逛街

小Q在周末的时候和他的小伙伴来到大城市逛街，一条步行街上有很多高楼，共有n座高楼排成一行。

小Q从第一栋一直走到了最后一栋，小Q从来都没有见到这么多的楼，所以他想知道他在每栋楼的位置处能看到多少栋楼呢？（当前面的楼的高度大于等于后面的楼时，后面的楼将被挡住）

使用两个单调递减栈分别从左边和右边进行遍历，不断弹出比当前元素小的栈中元素（因为会被遮挡），栈中的元素个数就是所能看到的楼数。

  • 时间复杂度：O()
  • 空间复杂度：O()
#include <vector>
class Solution {
public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 
     * @param heights int整型vector 
     * @return int整型vector
     */
    vector<int> findBuilding(vector<int>& heights) {
        // write code here
        stack<int> left,right;
        vector<int> ans(heights.size(),0);
        for(int i=0;i<heights.size();++i){
            ans[i]+=1+left.size();
            while(!left.empty()&&heights[i]>=heights[left.top()]){
                left.pop();
            }
            left.push(i);
        }
        
        for(int i=heights.size()-1;i>=0;--i){
            ans[i]+=right.size();
            while(!right.empty()&&heights[i]>=heights[right.top()]){
                right.pop();
            }
            right.push(i);
        }
        return ans;
        
    }
};

3：逆序对

作为程序员的小Q，他的数列和其他人的不太一样，他有2^n2n个数。

老板问了小Q一共 m次，每次给出一个整数q_i (1 <= i <= m)q**i(1<=i<=m), 要求小Q把这些数每2^{q_i}2q**i分为一组，然后把每组进行翻转，小Q想知道每次操作后整个序列中的逆序对个数是多少呢？

参考

可以使用归并排序找到所有的逆序对，其时间复杂度为nlogn，但仍会超时，利用归并招逆序对个数见此。可以注意到每次翻转只会造成归并时每一层的逆序对数和顺序对数对调，因此记录归并时每一层的逆序对和顺序对个数。代码改了很久还是不能全过。。

  • 时间复杂度：O()
  • 空间复杂度：O()
#include <iostream>
#include <vector>
#include <stack>
#include <algorithm>
#include <math.h>
using namespace std;


void merge(vector<int>& nums, int s1, int e1, int s2, int e2,
           vector<long long>& count) {      
    // cout<<s1<<" "<<e1<<" "<<s2<<' '<<e2<<endl;      
    if (e2 - s1 <= 0) return;    
    int s = s1, e = e2;
    vector<int> tmp(e2 - s1);
    long long neg = 0, posi = 0;
    int idx=0;
    while (s1 <= e1 && s2 <= e2) {
        if (nums[s1] > nums[s2]) {
            neg += e1 - s1 + 1;
            tmp[idx++] = nums[s2];
            ++s2;
        } else {
            posi += e2 - s2 + 1;
            tmp[idx++] = nums[s1];
            ++s1;
        }
    }
    count[0] += neg;
    count[1] += posi;
    while (s1 <= e1) {
        tmp[idx++] = nums[s1++];
    }
    while (s2 <= e2) {
        tmp[idx++] = nums[s2++];
    }
    for (int i = s, j = 0; i <= e; ++i, ++j) {
        nums[i] = tmp[j];
    }
}

vector<vector<long long>> layers;

void calLayerCount(vector<int> nums) {
    int n = nums.size();
    for (int len = 1; len < n; len *= 2) {
        vector<long long> count{0, 0};
        for (int i = 0; i < n;i+=2*len) {
            merge(nums, i, min(i + len - 1, n - 1), i + len, min(i + 2 * len - 1, n - 1),
                  count);
        }
        layers.push_back(count);
    }
}

int main() {
    int n;
    while (cin >> n) { // 注意 while 处理多个 case
        n = pow(2, n);
        vector<int> nums(n);
        for (int i = 0; i < n; ++i) {
            cin >> nums[i];
        }
        int m;
        cin >> m;
        int ques;
        calLayerCount(nums);
        for (int i = 0; i < m; ++i) {
            cin >> ques;
            long long ans = 0;
            for (int i = 0; i < ques; ++i) {
                swap(layers[i][0], layers[i][1]);
            }
            for (int i = 0; i < layers.size(); ++i) {
                ans += layers[i][0];
            }
            cout << ans << endl;
        }

    }
}

4：假期

由于业绩优秀，公司给小Q放了 n 天的假，身为工作狂的小Q打算在在假期中工作、锻炼或者休息。他有个奇怪的习惯：不会连续两天工作或锻炼。只有当公司营业时，小Q才能去工作，只有当健身房营业时，小Q才能去健身，小Q一天只能干一件事。给出假期中公司，健身房的营业情况，求小Q最少需要休息几天。

动态规划。

使用dp[i] [0] 记录第i天去工作最大的不休息天数，dp[i] [1]表示第i天去锻炼最大的不休息次数。

如果第i-1天不营业，那么第i天能否工作就与前一天 无关，即d[i] [0]=dp[i-1] [0] +nums1[i]。如果前一天营业，那么就等于前一天。最后，应与dp[i-1] [1] +nums1[i]比较，取最大值。 dp[i] [1]同理。

  • 时间复杂度：O()
  • 空间复杂度：O()
#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n;
    while (cin >> n) { // 注意 while 处理多个 case
        vector<int> nums1(n + 1), nums2(n + 1);
        nums1.push_back(0);
        nums2.push_back(0);
        for (int i = 1; i <= n; ++i) cin >> nums1[i];
        for (int i = 1; i <= n; ++i) cin >> nums2[i];
        vector<vector<int>> dp(n + 1, vector<int>(2, 0));
        for (int i = 1; i <= n; ++i) {
            if (nums1[i - 1] == 0) dp[i][0] = dp[i - 1][0] + nums1[i];
            else dp[i][0] = dp[i - 1][0];
            dp[i][0] = max(dp[i][0], dp[i - 1][1] + nums1[i]);
            if (nums2[i - 1] == 0) dp[i][1] = dp[i - 1][1] + nums2[i];
            else dp[i][1] = dp[i - 1][1];
            dp[i][1] = max(dp[i][1], dp[i - 1][0] + nums2[i]);
            //或者简化
            // dp[i][0] = max(dp[i - 1][0], dp[i - 1][1] + nums1[i]);
            // dp[i][1] = max(dp[i - 1][1], dp[i - 1][0] + nums2[i]);
        }
        cout << n - max(dp[n][0], dp[n][1]) << endl;
    }
}
// 64 位输出请用 printf("%lld")

5：视野争夺

小Q在进行一场竞技游戏,这场游戏的胜负关键就在于能否能争夺一条长度为L的河道,即可以看作是[0,L]的一条数轴。

这款竞技游戏当中有n个可以提供视野的道具−真视守卫,第i个真视守卫能够覆盖区间[xi,yi]。现在小Q想知道至少用几个真视守卫就可以覆盖整段河道。

贪心。

先排序，不断找以当前区间为起点，所能到达的最大范围。

#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    int n,l;
    while (cin >> n >> l) { // 注意 while 处理多个 case
        vector<vector<int>> nums(n,vector<int>(2,0));
        for(int i=0;i<n;++i) cin>>nums[i][0]>>nums[i][1];
        sort(nums.begin(),nums.end(),[](vector<int> &a,vector<int> &b){
            return a[0]<b[0]||(a[0]==b[0]&&a[1]<b[1]);
        });
        // for(auto tmp:nums) cout<<tmp[0]<<" "<<tmp[1]<<endl;
        int ans=0,curRight=0,curLeft=0;
        int i=0;
        while(i<nums.size()){
            int lastRight=curRight;
            while(i<nums.size()&&nums[i][0]<=lastRight){
                curRight=max(curRight,nums[i][1]);
                ++i;
            }
            ++ans;
            if(curRight>=l) break;
            if(i>=nums.size()){
                ans=-1;
                break;
            }
        }

        cout<<ans<<endl;
        
    }
}
